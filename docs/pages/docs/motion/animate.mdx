```tsx
import { c } from "@marathontvapp/carnation";
import { animate, m, useMotionValue } from "@marathontvapp/carnation/motion";

function Component() {
  const translateX = useMotionValue(0);

  return (
    <c.div>
      <m.div style={{ translateX }} />

      <c.button
        onPress={() => {
          animate(translateX, Math.random() * 100, {
            type: "timing",
            ease: "easeInOut",
            duration: 2,
          });
        }}
      >
        <c.span>Move</c.span>
      </c.button>
    </c.div>
  );
}
```
