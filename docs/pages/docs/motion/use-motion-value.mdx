```tsx
import { c, m, useMotionValue } from "carnation-ds";

function Component() {
  const translateX = useMotionValue(0);

  return (
    <c.div>
      <m.div style={{ translateX }} />

      <c.button
        onPress={() => {
          translateX.set(Math.random() * 100);
        }}
      >
        <c.span>Move</c.span>
      </c.button>
    </c.div>
  );
}
```
